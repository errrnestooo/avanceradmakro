    import numpy as np
import matplotlib.pyplot as plt

def tent_map(x, n=5):
    """iterate a tent map with x as the initial condition in the domain of x in (0,1), with n steps (default n=5)"""
    inputs = np.array([])
    outputs = np.array([])
    print(x)
    for i in range(1, n):
        inputs = np.append(inputs, x)
        if x < 0.5:
            x = 2 * x
        elif x > 0.5:
            x = 2 - 2 * x
        
        outputs = np.append(outputs, x)
        print(x)
        plt.plot(inputs, outputs)
